CREATE DATABASE CBB;

USE CBB;

CREATE TABLE Cervezas (
  Marca Varchar(30) NOT NULL PRIMARY KEY,
  Color Varchar(20),
  Tamanio INT,
  Precio FLOAT,
  Cia Varchar(20),
  gGL FLOAT
) ENGINE=InnoDB;

CREATE TABLE Bares (
  NomBar Varchar(30) NOT NULL PRIMARY KEY,
  Zona Varchar(20),
  TipoBar Varchar(20)
) ENGINE=InnoDB;

CREATE TABLE Borrachos (
  NomBorr Varchar(30) NOT NULL PRIMARY KEY,
  Edad INT,
  TipoBorr Varchar(20),
  Profesion Varchar(10),
  Sexo Varchar(2)
) ENGINE=InnoDB;

CREATE TABLE Asisten (
  NomBorr Varchar(30) NOT NULL,
  NomBar Varchar(30) NOT NULL,
  PRIMARY KEY (NomBorr, NomBar),
  FOREIGN KEY (NomBorr) REFERENCES Borrachos (NomBorr),
  FOREIGN KEY (NomBar) REFERENCES Bares (NomBar)
  ON DELETE CASCADE
  ON UPDATE CASCADE
) ENGINE=InnoDB;

CREATE TABLE Consumen (
  Marca Varchar(30) NOT NULL,
  NomBorr Varchar(30) NOT NULL,
  PRIMARY KEY (Marca, NomBorr),
  FOREIGN KEY (Marca) REFERENCES Cervezas (Marca),
  FOREIGN KEY (NomBorr) REFERENCES Borrachos (NomBorr)
  ON DELETE CASCADE
  ON UPDATE CASCADE
) ENGINE=InnoDB;

cREATE TABLE Venden (
  Marca Varchar(30) NOT NULL,
  NomBar Varchar(30) NOT NULL,
  PRIMARY KEY (Marca, NomBar),
  FOREIGN KEY (Marca) REFERENCES Cervezas (Marca),
  FOREIGN KEY (NomBar) REFERENCES Bares (NomBar)
  ON DELETE CASCADE
  ON UPDATE CASCADE
) ENGINE=InnoDB;
